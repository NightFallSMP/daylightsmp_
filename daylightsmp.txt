        function showPage(pageName) {
            const pages = document.querySelectorAll('.page');
            pages.forEach(page => {
                page.classList.remove('active');
            });
            document.getElementById(pageName).classList.add('active');
            window.scrollTo(0, 0);
        }

        function toggleFeature(card) {
            card.classList.toggle('expanded');

        // Toggle rule explanations with modal
        function toggleRule(rule) {
            const title = rule.querySelector('.rule-header').textContent;
            const explanation = rule.querySelector('.rule-explanation p').textContent;
            
            document.getElementById('modalTitle').textContent = title;
            document.getElementById('modalText').textContent = explanation;
            document.getElementById('ruleModal').classList.add('active');
        }

        // Close modal
        function closeModal(event) {
            if (event && event.target.className !== 'modal-overlay') return;
            document.getElementById('ruleModal').classList.remove('active');
        }

        // Close modal with ESC key
        document.addEventListener('keydown', function(e) {
            if (e.key === 'Escape') {
                closeModal();
            }
        });
